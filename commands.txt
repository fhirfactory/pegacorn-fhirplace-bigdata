# Build base image
docker build --rm --build-arg IMAGE_BUILD_TIMESTAMP="%date% %time%" -t pegacorn-base-hadoop:1.0.0 --file Dockerfile .

# Deploy Master Node (NameNode)
docker build --rm --build-arg IMAGE_BUILD_TIMESTAMP="%date% %time%" -t pegacorn-fhirplace-namenode:1.0.0-snapshot --file Dockerfile .
\helm\helm upgrade pegacorn-fhirplace-namenode-site-a --install --namespace site-a --set serviceName=pegacorn-fhirplace-namenode,hostPathNameDir=/data/hadoop-name,clusterName=Pegacorn,imageTag=1.0.0-snapshot,numOfPods=1 helm


# Deploy Worker Nodes (DataNodes)
docker build --rm --build-arg IMAGE_BUILD_TIMESTAMP="%date% %time%" -t pegacorn-fhirplace-datanode:1.0.0-snapshot --file Dockerfile .

# Retrieve CLUSTER-IP of NameNode service
kubectl get svc pegacorn-fhirplace-namenode -n site-a --no-headers | awk {'print $3'}

\helm\helm upgrade pegacorn-fhirplace-datanode-alpha-site-a --install --namespace site-a --set serviceName=pegacorn-fhirplace-datanode-alpha,hostPathDataDir=/data/hadoop-data-alpha,clusterIP=10.98.38.87,imageTag=1.0.0-snapshot,numOfPods=1 helm
\helm\helm upgrade pegacorn-fhirplace-datanode-beta-site-a --install --namespace site-a --set serviceName=pegacorn-fhirplace-datanode-beta,hostPathDataDir=/data/hadoop-data-beta,clusterIP=10.98.38.87,imageTag=1.0.0-snapshot,numOfPods=1 helm
